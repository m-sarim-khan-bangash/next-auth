go to github => githcom.com/settings/developers create
register 0auth app
run command npx auth secret
create .env.local file
add AUTH_SECRET to .env.local file => npx auth secret to generate AuthSecret
copy client id and secret to .env.local file
go to src/app/api/auth/[...nextauth]/route.js
add import { handlers } from "@/auth";
export const { GET, POST } = handlers;
go to next.config.mjs
add images: { domains: ["avatars.githubusercontent.com"] }
